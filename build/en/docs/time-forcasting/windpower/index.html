LSTM训练​

def prepare_set(df,percentage):
    train_size = int(len(df) * percentage)
    train_set = df[:train_size].copy()
    test_set = df[train_size:].copy()
    return train_set,test_set

import torch
import torch.nn as nn

#拟根据方差贡献率选取IMF5-10简化训练难度 ，然后通过权重调整的方式预测

# 定义LSTM模型
class LSTMf(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers):
        super(LSTMf, self).__init__()
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size, 1)  # 预测下一个数据

    def forward(self, x):
        # 初始化隐藏状态和细胞状态
        h0 = torch.zeros(self.num_layers, x.size(0), self.hidden_size).to(x.device)
        c0 = torch.zeros(self.num_layers, x.size(0), self.hidden_size).to(x.device)
        
        # 前向传播LSTM
        out, _ = self.lstm(x, (h0, c0))
        
        # 取最后一个时间步的输出
        out = out[:, -1, :]
        
        # 通过全连接层进行预测
        out = self.fc(out)

        return out  
   
import torch.optim as optim
import pandas as pd
import matplotlib.pyplot as plt

emd_data = pd.read_csv('CEEMDAN2.csv')
train_x,test_x = prepare_set(emd_data,0.7)

#IMF5-7复杂，选用双层的lstm，其他简单的单层就好
# lstms = []
# for _ in range(3):
#     lstms.append(LSTMf(1,28,2))
# for _ in range(3):
#     lstms.append(LSTMf(1,30,1))
lstm1 = LSTMf(1,32,1)

# 定义损失函数和优化器
criterion = nn.MSELoss()  # 均方误差损失
optimizer = optim.Adam(lstm1.parameters(), lr=0.003)  # Adam优化器

batch_size = 16
epochs = 30
time_step = 10
iter_nums = emd_data.shape[0]//(batch_size*time_step)

epoch_losses = []
for epoch in range(epochs):
    epoch_loss = 0
    #划分训练批次batch_size*time_step*input_size,这里选取最简单的IMF10
    for iter_num in range(iter_nums):
        X = emd_data.iloc[iter_num*batch_size*time_step:(iter_num+1)*batch_size*time_step,9].values
        Y = []
        for i in range(0, time_step * batch_size, time_step):
            Y.append(emd_data.iloc[iter_num*batch_size*time_step+i+time_step,9])
        # print(X)
        # print(Y)
        Y = torch.tensor(Y,dtype=torch.float32).reshape(batch_size,1)
        X = torch.tensor(X,dtype=torch.float32).reshape(batch_size,time_step,1)
        output = lstm1(X)
        loss = criterion(output,Y)
        print(f"epoch:{epoch+1},第{iter_num+1}批次,损失为{loss}")
        epoch_loss += loss.item()
        optimizer.zero_grad()  # 清空之前的梯度
        loss.backward()  # 反向传播，计算梯度
        optimizer.step()  # 更新参数
    epoch_losses.append(epoch_loss / iter_nums)

# 绘制损失曲线
plt.plot(epoch_losses, label='Training Loss')
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.title('Training Loss Curve')
plt.legend()
plt.show()
